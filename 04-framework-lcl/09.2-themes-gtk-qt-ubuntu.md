function GetUbuntuVersion: string;  
var
  ReleaseFile: TStringList;
  Line: string;
begin
  Result := 'Unknown';

  if FileExists('/etc/lsb-release') then
  begin
    ReleaseFile := TStringList.Create;

        end;

    end;
  end;
end;
